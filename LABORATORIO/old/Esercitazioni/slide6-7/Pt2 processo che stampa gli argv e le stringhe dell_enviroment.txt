/* echoall.c */ 
#include <stdio.h> 
#include <stdlib.h>

int main(int argc, char * argv[]){ 
int	i;
char	**ptr; 
extern char **environ;

for (i=0; i < argc; i++) // echo di tutti gli arg da riga di cmd 
printf(“argv[%d]: %s\n”, i, argv[i]);

for (ptr = environ; *ptr !=0; ptr++) /* echo stringhe di env */ 
printf(“%s\n”, *ptr);
exit(0);
}
